def find_longest_word(string):
    words = string.split()
    longest_word = ""
    max_length = 0
    for word in words:
        if len(word) > max_length:
            max_length = len(word)
            longest_word = word
    return longest_word

def count_character_frequency(string, char):
    count = 0
    for c in string:
        if c == char:
            count += 1
    return count

def is_palindrome(string):
    string1 = string[::-1]
    if(string1==string):
        print("The string is a palindrome.")
    else:
        print("The string is not a palindrome.")

def find_substring_index(string, substring):
    for i in range(len(string) - len(substring) + 1):
        if string[i:i+len(substring)] == substring:
            return i
    return -1

def count_word_occurrences(string):
    words = string.split()
    word_count = {}
    for word in words:
        if word in word_count:
            word_count[word] += 1
        else:
            word_count[word] = 1
    return word_count


print("\na) To display word with the longest length \nb) To determines the frequency of occurrence of particular character in the string\nc) To check whether given string is palindrome or not\nd) To display index of first appearance of the substring\ne) To count the occurrences of each word in a given string")
operation = input("\nEnter the operation (a, b, c, d, e): ")

if operation == 'a':

        print(f"Frequency of '{char_to_count}': {char_frequency}")
elif operation == 'c':
        input_string = input("Enter a string: ")

elif operation == 'd':
        input_string = input("Enter a string: ")
